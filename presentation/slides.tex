\begin{frame}{The problem}
  \begin{columns}[T]
    \begin{column}{0.58\textwidth}
      \begin{itemize}
        \item Public archives (PRIDE, MassIVE) hold \textbf{thousands of DIA datasets}.
        \item Each was processed with a \textbf{different software configuration} $\rightarrow$ analytically siloed.
        \item \diann{} is the dominant DIA engine, but most deployments are
              \textbf{artisanal}: bespoke scripts, hand-curated sheets,
              hard-coded versions.
        \item New \diann{} versions quantify more precursors and proteins
              every release $\rightarrow$ reanalysis is \textbf{not a one-time event}.
      \end{itemize}
      \vspace{0.6em}
      \begin{block}{Goal}
        One reproducible, metadata-driven workflow that scales \diann{} to
        archive-sized cohorts.
      \end{block}
    \end{column}
    \begin{column}{0.42\textwidth}
      \centering
      \includegraphics[width=\linewidth,height=0.72\textheight,keepaspectratio]{manuscript/workflow_panel}
      \\[-0.2em]
      {\scriptsize\color{qmgrey}The \quantmsdiann{} workflow}
    \end{column}
  \end{columns}
\end{frame}

\section{SDRF extensions}

\begin{frame}{SDRF as the single source of truth}
  \quantmsdiann{} takes \textbf{one \sdrf{} file} as primary input and derives
  every \diann{} parameter from it --- removing the most common source of
  analysis-script divergence between labs.
  \vspace{0.5em}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Sample annotation (ontology-validated)}
      \begin{itemize}\small
        \item \texttt{characteristics[cell line]}
        \item \texttt{characteristics[cellosaurus accession]}
        \item \texttt{characteristics[organism part]}, \texttt{disease}
        \item \texttt{factor value[\ldots]}
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Acquisition $\rightarrow$ \diann{} config}
      \begin{itemize}\small
        \item \texttt{comment[instrument]}
        \item \texttt{comment[proteomics data acquisition method]}
        \item \texttt{comment[precursor/fragment mass tolerance]}
        \item \texttt{comment[modification parameters]}
        \item \texttt{comment[cleavage agent details]}
        \item \texttt{comment[file uri]} / \texttt{[associated file uri]}
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{SDRF extensions added for DIA}
  New / DIA-specific columns let one SDRF drive the whole run:
  \vspace{0.4em}
  \begin{block}{Example row (PXD004701, Sun 2023)}
    \begin{semiverbatim}\scriptsize
characteristics[cellosaurus accession]  = CVCL_3040
comment[instrument]                     = NT=TripleTOF 5600;AC=MS:1000932
comment[proteomics data acquisition method] = Data-independent acquisition
comment[modification parameters]        = NT=Carbamidomethyl;MT=fixed;TA=C
comment[modification parameters]        = NT=Oxidation;MT=variable;TA=M
comment[cleavage agent details]         = NT=Trypsin/P;AC=MS:1001313
comment[precursor mass tolerance]       = 20 ppm
comment[fragment mass tolerance]        = 20 ppm
comment[file uri]                       = .../PXD004701/guot_..._BR1.wiff
    \end{semiverbatim}
  \end{block}
  \begin{itemize}\small
    \item \textbf{Cellosaurus accession} $\rightarrow$ unambiguous cell-line identity, queryable downstream.
    \item \textbf{Per-run tolerances \& mods} $\rightarrow$ propagated to per-file \diann{} invocations.
    \item \textbf{File URIs} $\rightarrow$ raw files fetched directly from PRIDE/MassIVE (\texttt{pridepy}).
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Encoding plexDIA in SDRF}
  We extended the \sdrf{} representation so that \textbf{plexDIA / mTRAQ
  multiplexing} is described entirely in metadata --- no side-car channel
  files.
  \vspace{0.3em}
  \begin{block}{One row per channel, sharing the same raw file}
    \begin{semiverbatim}\scriptsize
source name      comment[data file]  comment[label]  factor value[condition]
Sample_A_rep1    wJD803.raw          \textcolor{qmblue}{MTRAQ0}          Sample_A
Sample_B_rep1    wJD803.raw          \textcolor{qmblue}{MTRAQ4}          Sample_B
Sample_C_rep1    wJD803.raw          \textcolor{qmblue}{MTRAQ8}          Sample_C
    \end{semiverbatim}
  \end{block}
  \begin{itemize}\small
    \item \textbf{Channel labels} (\texttt{MTRAQ0/4/8}) live in
          \texttt{comment[label]}; the shared \texttt{comment[data file]}
          makes the \textbf{channel\,$\rightarrow$\,sample} mapping explicit.
    \item The label reagents are resolved to ontology terms and turned into
          \diann{}'s plexDIA channel configuration automatically.
    \item Same mechanism covers SILAC-style non-isobaric labels --- the
          acquisition method stays \texttt{data-independent acquisition}.
  \end{itemize}
\end{frame}

\begin{frame}{From SDRF to outputs}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Runtime selectable}
      \begin{itemize}\small
        \item Container-pinned \diann{} 1.8.1, 2.1.0, 2.2.0, 2.3.2, 2.5.0
              (\texttt{--diann\_version}).
        \item Same SDRF $\rightarrow$ many \diann{} versions in one invocation.
      \end{itemize}
      \vspace{0.4em}
      \textbf{Vendor formats}
      \begin{itemize}\small
        \item Thermo \texttt{.raw}, Bruker \texttt{.d} --- native to current \diann{}.
        \item Sciex \texttt{.wiff/.wiff2} --- via \texttt{wiffconverter}.
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Harmonised outputs}
      \begin{itemize}\small
        \item Standard \diann{} report tables.
        \item MSstats input.
        \item \textcolor{qmblue}{\textsc{qpx}} (Quantitative Proteomics
              eXchange) Parquet archive --- queryable with DuckDB, MuData
              export for scverse.
        \item \texttt{pmultiqc} QC report.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Summary}
  \begin{itemize}
    \item \quantmsdiann{}: an \textbf{independent, SDRF-driven} Nextflow/nf-core
          workflow dedicated to \diann{}.
    \item \textbf{SDRF extensions} carry per-run acquisition parameters and
          ontology-validated sample identity into every \diann{} invocation
          and into the outputs.
    \item \textbf{Benchmarks}: matches the standalone \diann{} GUI on
          ProteoBench across 5 versions $\times$ 4 modalities.
    \item \textbf{Results}: recovers comparable-or-deeper coverage on public
          cell-line deposits; scales a 2{,}300-run cohort to 2.4 h.
  \end{itemize}
  \vspace{0.8em}

\end{frame}
